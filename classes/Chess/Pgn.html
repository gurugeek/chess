def load(filename)
  data = File.open(filename, 'r').read
  data.gsub!(%r\{.*?\}/, '')
  TAGS.each do |t|
    instance_variable_set("@#{t}", data.match(%r^\[#{t.capitalize} ".*"\]\s?$/).to_s.strip[t.size+3..-3])
  end
  game_index = data.index(%r^1\./)
  raise Chess::InvalidPgnFormatError.new(filename) if game_index.nil?
  game = data[game_index..-1].strip
  @moves = game.gsub("\n", ' ').split(%r\d+\./).collect{|t| t.strip}[1..-1].collect{|t| t.split(' ')}.flatten
  @moves.delete_at(@moves.size-1) if @moves.last =~ %r(0\-1)|(1\-0)|(1\/2)|(1\/2\-1\/2)|(\*)/
  return self
end

def to_s
  s = ''
  TAGS.each do |t|
    tag = instance_variable_get("@#{t}")
    s << "[#{t.capitalize} \"#{tag}\"]\n"
  end
  s << "\n"
  m = ''
  @moves.each_with_index do |move, i|
    m << "#{i/2+1}. " if i % 2 == 0
    m << "#{move} "
  end
  m << @result unless @result.nil?
  s << m.gsub(%r(.{1,78})(?: +|$)\n?|(.{78})/, "\\1\\2\n")
end
